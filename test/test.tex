\section{公式}


\begin{equation}
	\bm{F} = m\bm{a}.
	\label{eq:1}
\end{equation}
\cref{eq:1}用到了\texttt{bm}包，可以方便地加粗符号。

\begin{equation}
	\int^b_a f(x) \dif x=F(x)\bigg|^b_a.
	\label{eq:2}
\end{equation}
\cref{eq:2}用到了\texttt{commath}包，可以方便地写微分算符。

\begin{equation}
	\me ^{\pi \mi} + 1 = 0.
	\label{eq:3}
\end{equation}
\cref{eq:3}用到了自定义的\verb|\me, \mi|来表示常数$\me,\ \mi$.
